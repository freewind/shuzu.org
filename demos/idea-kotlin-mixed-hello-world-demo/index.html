Kotlin Mixed (JVM + JavaScript) Hello World Demo ================================================ This project is created by IDEA directly, chose `kotlin - mixed` Dirs: - `src`: common kotlin code for both JVM and JS module - `kotlin-mixed-hello-world-demo-js`: JS module - `kotlin-mixed-hello-world-demo-jvm`: JVM module
kotlin-mixed-hello-world-demo-js/ build.gradle 
buildscript {
    ext.kotlin_version = '1.2.31'

    repositories {
        mavenCentral()
    }
    dependencies {
        classpath "org.jetbrains.kotlin:kotlin-gradle-plugin:$kotlin_version"
    }
}
apply plugin: 'kotlin-platform-js'

repositories {
    mavenCentral()
}

dependencies {
    compile "org.jetbrains.kotlin:kotlin-stdlib-js:$kotlin_version"
    expectedBy project(":")
    testCompile "org.jetbrains.kotlin:kotlin-test-js:$kotlin_version"
}
kotlin-mixed-hello-world-demo-js/src/main/kotlin/ HelloJs.kt 
fun main(args: Array<String>) {
    hello("JS")
}
