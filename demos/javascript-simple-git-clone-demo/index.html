JavaScript "simple-git" Clone Demo ================================== 使用"simple-git"将指定的git仓库clone到本地 ``` npm install node run demo ``` 然后就会在`local-repo`目录下看到新clone下来的`demo-project`。
package.json 
{
  "scripts": {
    "demo": "mkdir local-repo && node demo.js"
  },
  "dependencies": {
    "simple-git": "^1.96.0"
  }
}
demo.js 
const workingDirPath = './local-repo'
const simpleGit = require('simple-git')(workingDirPath)

simpleGit.clone('https://github.com/freewind-demos/javascript-frontend-hello-world-demo.git', 'demo-project')

console.log('cloned to: local-repo/demo-project')
