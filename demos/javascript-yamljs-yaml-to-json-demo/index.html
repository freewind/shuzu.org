JavaScript Yamljs Yaml to Json Demo =================================== ``` npm install npm run demo ``` It will print: ``` Hello, Javascript ```
yaml-to-json.js 
const YAML = require('yamljs')

const yamlObj = YAML.load('hello-yaml.yaml')
console.log('-------------- parsed yaml object -------------')
console.log(yamlObj)

const json = JSON.stringify(yamlObj, null, 2)
console.log('-------------- converted JSON --------------')
console.log(json)
hello-json.json 
{
  "name": "YAML & JSON",
  "keywords": [
    "JavaScript",
    "YamlJS",
    {
      "formats": [
        "YAML",
        "JSON"
      ]
    }
  ]
}
json-to-yaml.js 
const YAML = require('yamljs')
const fs = require('fs')

const jsonStr = fs.readFileSync('hello-json.json', 'utf-8')
console.log('--------------- JSON str ------------------')
console.log(jsonStr)

const jsonObj = JSON.parse(jsonStr)
console.log('--------------- JSON obj -------------------')
console.log(jsonObj)

const yaml = YAML.dump(jsonObj)
console.log('------------- converted YAML --------------')
console.log(yaml)
hello-yaml.yaml

  - YamlJS
  - formats:
