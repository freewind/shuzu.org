jasmine-debug-in-webstorm
Jasmine Debugging Demo ====================== How to use webstorm to debug jasmine specs. ``` npm install -g jasmine npm install ``` Then set some break points, you can single debug in webstorm by running `run-specs.js` directly in debugging mode.
hello-spec.js
spec/hello-spec.js
var hello = require('../src/hello');

describe("A suite", function () {
  it("contains spec with an expectation", function () {
    expect(hello()).toBe("Hello,world!");
  });
});
jasmine.json
spec/support/jasmine.json
{
  "spec_dir": "spec",
  "spec_files": [
    "**/*[sS]pec.js"
  ],
  "helpers": [
    "helpers/**/*.js"
  ],
  "stopSpecOnExpectationFailure": false,
  "random": false
}
core.js
src/core.js
module.exports = function() {
  console.log("core function");
}
hello.js
src/hello.js
let core = require('./core');

module.exports = function () {
  core();
  let h = "Hello,";
  let w = "world!";
  return h + w;
};
package.json
package.json
{
  "name": "jasmine-debug-in-webstorm",
  "version": "1.0.0",

    "test": "jasmine-node --verbose --junitreport spec"

  "keywords": [],
  "author": "",
  "license": "ISC",
  "devDependencies": {
    "jasmine": "^2.4.1"

}
run-spec.js
run-spec.js
var Jasmine = require('jasmine');
var jasmine = new Jasmine();

jasmine.loadConfigFile('spec/support/jasmine.json');

jasmine.execute();
