jasmine-debug-in-webstorm
Jasmine Debugging Demo ====================== How to use webstorm to debug jasmine specs. ``` npm install -g jasmine npm install ``` Then set some break points, you can single debug in webstorm by running `run-specs.js` directly in debugging mode.
run-spec.js
run-spec.js
var Jasmine = require('jasmine');
var jasmine = new Jasmine();

jasmine.loadConfigFile('spec/support/jasmine.json');

jasmine.execute();
hello-spec.js
spec/hello-spec.js
var hello = require('../src/hello');

describe("A suite", function () {
  it("contains spec with an expectation", function () {
    expect(hello()).toBe("Hello,world!");
  });
});
