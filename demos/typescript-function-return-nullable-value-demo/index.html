TypeScript Function Returns Nullable Value Demo =============================================== Define a function which returns nullable value. ``` npm install npx ts-node hello.ts ```
hello.ts
hello.ts
function hello(name: string): string | null {
    if (name.length === 0) {
        return null
    } else {
        return `Hello, ${name}!`
    }
}

console.log(hello('typescript'))
console.log(hello(''))
