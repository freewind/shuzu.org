Webpack Css Loader Config Demo ============================== ``` npm install -g webpack npm install webpack ``` Then open `index.html`.
entry.js
entry.js
require('./index.css');

document.write('it works');
index.css
index.css
body {
  background-color: yellow;
}
webpack.config.js
webpack.config.js
module.exports = {
  entry: './entry.js',
  output: {
    path: __dirname,
    filename: 'bundle.js'
  },
  module: {
    loaders: [{ 
      test: /\.css$/,
      loader: 'style!css' // note style is before css
    }]
  }
}
