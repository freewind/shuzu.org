Ant Java Demo ============== Compile, Run, and Jar ``` brew install ant ``` Run: ``` ant ``` or ``` ant run ``` You will see: ``` compile: [javac] /Users/freewind/workspace/ant-javac/build.xml:9: warning: 'includeantruntime' was not set, defaulting to build.sysclasspath=last; set to false for repeatable builds [javac] Compiling 1 source file to /Users/freewind/workspace/ant-javac/build/classes run: [java] Hello, java BUILD SUCCESSFUL Total time: 0 seconds ``` Package a `jar` file and run it: ``` java jar ``` Then: ``` java -jar build/hello.jar ```
build.xml
build.xml
<?xml version="1.0" ?>
<project name="ant-demo" basedir="." default="run">
	<target name="clean">
		<delete dir="${basedir}/build" />
	</target>

	<target name="compile">
		<mkdir dir="${basedir}/build/classes" />
		<javac srcdir="${basedir}" destdir="${basedir}/build/classes" />
	</target>

	<target name="run" depends="compile">
		<java classname="Hello">
			<classpath>
				<pathelement path="${basedir}/build/classes" />
			</classpath>
		</java>
	</target>

	<target name="jar" depends="run">
		<jar destfile="build/hello.jar" basedir="${basedir}/build/classes">
			<manifest>
				<attribute name="Main-class" value="Hello" />
			</manifest>
		</jar> 
	</target>
</project>
