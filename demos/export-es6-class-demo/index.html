Es6 Class Exporting Demo ======================== ``` node index.js ``` (`node -v` should `>=6`)
app.js
app.js
let Core1 = require('./core1');
let {Core2} = require('./core2');

let core1 = new Core1();
let core2 = new Core2();

core1.hello();
core2.hello();
core1.js
core1.js
class Core1 {
  hello() {
    console.log('hello from core1');
  }
}

module.exports = Core1;
core2.js
core2.js
class Core2 {
  hello() {
    console.log('hello from core2');
  }
}

module.exports = {
  Core2 : Core2
};
index.js
index.js
let Core1 = require('./core1');
let {Core2} = require('./core2');

let core1 = new Core1();
let core2 = new Core2();

core1.hello();
core2.hello();
