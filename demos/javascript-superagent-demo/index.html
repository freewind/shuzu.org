JavaScript SuperAgent Demo ========================== ``` npm install node index.js ``` You will get a big JSON about the response.
index.js 
const request = require('superagent')

request.get('http://taobao.com')
    .end((err, res) => {
        if (err) return console.error(err)
        console.log(res)
    })
package.json 
{
  "dependencies": {
    "superagent": "^3.8.3"
  }
}
