Gradle Test ========== Use [fest](https://github.com/alexruiz/fest-assert-2.x) to do assertions in tests. (Requires JDK 1.8) ``` ./gradlew test ``` If there is any failure test, you can see more details: ``` ./gradlew --info test ```
build.gradle
build.gradle
apply plugin: 'java'

repositories {
    mavenLocal()
    jcenter() // fast!
    mavenCentral() // very slow
}

dependencies {
    testCompile(
            'junit:junit:4.11',
            'org.assertj:assertj-joda-time:2.0.0'
    )
}
Hello.java
src/main/java/my/Hello.java
package my;

public class Hello {

    public static String hello(String name) {
        return "Hello, " + name;
    }

}
HelloSpec.java
src/test/java/my/HelloSpec.java
package my;

import org.junit.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class HelloSpec {

    @Test
    public void shouldSayHello() throws Exception {
        String greeting = Hello.hello("fest");
        assertThat(greeting).isEqualTo("Hello, fest");
    }

}
