package my

object Hello {

  def hello(name: String) = s"""Hello, $name"""

}
HelloSpec.scala
src/test/scala/my/HelloSpec.scala
package my

import org.specs2.mutable.Specification

class HelloSpec extends Specification {

  "Hello" should {
    "say hello to someone" in {
      Hello.hello("sbt") === "Hello, sbt"
    }
  }

}
