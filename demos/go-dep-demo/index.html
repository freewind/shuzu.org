Go Dep Demo =========== Use [dep](https://golang.github.io/dep) to manage dependency. ``` brew install dep dep ensure go run hello.go ``` Notice ------ For this project, I use command: ``` dep init dep ensure -add github.com/pkg/errors ```

package main

import (
	"fmt"
	"github.com/pkg/errors"
)

func main() {
	err := errors.New("my error")
	fmt.Println(err)
}
