Maven Run Application Demo ========================== Run main class from `mvn` command line: ``` mvn compile exec:java ``` You will see `Hello, Maven!` printed in console.
pom.xml 
<project xmlns="http://maven.apache.org/POM/4.0.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/maven-v4_0_0.xsd">
    <modelVersion>4.0.0</modelVersion>
    <groupId>github.freewind</groupId>
    <artifactId>maven-demo</artifactId>
    <version>1.0-SNAPSHOT</version>
    <name>maven-run-application-demo</name>

    <properties>
        <exec.mainClass>demo.Hello</exec.mainClass>
    </properties>

</project>
src/test/java/demo/ HelloTest.java 
package demo;

import org.junit.Test;

import static org.junit.Assert.assertEquals;

public class HelloTest {

    @Test
    public void shouldGreetingWithPassingName() throws Exception {
        Hello hello = new Hello();
        assertEquals(hello.greeting("Freewind"), "Hello, Freewind!");
    }

}
src/main/java/demo/ Hello.java 
package demo;

public class Hello {

    public static void main(String[] args) {
        System.out.println("Hello, Maven!");
    }
    
}
