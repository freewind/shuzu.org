gradle-jetty-web-demo
Simple JSP pages demo without `web.xml` ``` gradle jettyRun ```
build.gradle
build.gradle
apply plugin: 'java'
apply plugin: 'idea'
apply plugin: 'jetty'
apply plugin: 'war'


repositories {
    mavenCentral()
    maven {
        url "http://dl.bintray.com/greghaskins/maven"
    }
}

dependencies {
    compile(
            'org.apache.commons:commons-lang3:3.2.1',
            'org.apache.commons:commons-io:1.3.2',
            'com.google.guava:guava:16.0'
    )
    testCompile(
            'com.greghaskins:spectrum:0.4.4',
            'org.assertj:assertj-core:2.2.0',
            'junit:junit:4.11'
    )
}
HelloWorld.java
src/main/java/example/HelloWorld.java
package example;

public class HelloWorld {

    public String greeting() {
        return "Hello, world";
    }
}
HelloWorldSpec.java
src/test/java/example/HelloWorldSpec.java
package example;

import org.junit.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class HelloWorldSpec {

    @Test
    public void should_return_greeting_word() throws Exception {
        HelloWorld helloWorld = new HelloWorld();
        assertThat(helloWorld.greeting()).isEqualTo("Hello, world");
    }
}
