TypeScript Call Getter in Method Demo ===================================== Call getter in method will still compute the getter multi times. ``` npm install npx ts-node hello.ts ```
hello.ts
hello.ts
class Hello {
    name: string

    constructor(name: string) {
        this.name = name
    }

    get bigName(): string {
        console.log('--- compute bigName ---')
        return this.name.toUpperCase()
    }

    sayHello() {
        console.log('Hello, ' + this.bigName)
        console.log('Hello, ' + this.bigName)
        console.log('Hello, ' + this.bigName)
    }
}

new Hello('typescript').sayHello()
``` --- compute bigName --- Hello, TYPESCRIPT --- compute bigName --- Hello, TYPESCRIPT --- compute bigName --- Hello, TYPESCRIPT ```
