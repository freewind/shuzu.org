Express Blocking Handler ------------------------ If there are some slow blocking code in route handlers, what will happen. ``` npm install node index.js ``` Then open several terminal sessions, visit the url: <http://localhost:3000> at the near same time. You will see all the requests are accepted, but each later request has more time to get the response.
index.js
index.js
var express = require('express');
var sleep = require('sleep').sleep;

var app = express();

app.use(function(req, res, next) {
  console.log(req.path + ': ' + new Date);
  next();
});

app.get('/', function(req, res) {
  sleep(10);
  res.send('Hello');
});

var server = app.listen(3000, function () {
  var host = server.address().address;
  var port = server.address().port;

  console.log('Example app listening at http://%s:%s', host, port);
});
