Go Hello World Demo =================== ``` brew install go go get go run hello.go ``` You will see `Hello, go!`
hello.go 
package main

import "fmt"

type X interface{}
type A struct{}
type B struct{}

func main() {
	fmt.Println("Hello")
	var x X
	x = A{}
	b := x.(B) // will panic, good
	fmt.Println(b)
}
