Groovy Closure Demo =================== Groovy Closure is the most powerful feature of Groovy: 1. `this` 2. `owner` 3. `delegate` See the files under `src/main/groovy`.
build.gradle 
apply plugin: 'groovy'

repositories {
    mavenLocal()
    jcenter()
}

dependencies {
    compile 'org.codehaus.groovy:groovy-all:2.5.0'
    testCompile 'junit:junit:4.11'
    testCompile 'org.easytesting:fest-assert-core:2.0M10'
}

// used to generate `./gradlew`
// by run `gradle wrapper`
task wrapper(type: Wrapper) {
    gradleVersion = '4.8' //version required
}
src/main/groovy/my/ this-owner-delegate.groovy 
package my

class MyClass {

    def outerClosure = {

        println("----------- outerClosure() ------------")
        println this.class.name
        println owner.class.name
        println delegate.class.name

        def nestedClosure = {
            println("----------- nestedClosure() ------------")
            println this.class.name
            println owner.class.name
            println delegate.class.name
        }

        nestedClosure()
    }
}

def closure = new MyClass().outerClosure
closure()
src/main/groovy/my/ change-delegate.groovy 
package my

Closure hello = {
    println message
}

class Hello {
    String message = "Hello delegate!"
}

hello.delegate = new Hello()
hello()
