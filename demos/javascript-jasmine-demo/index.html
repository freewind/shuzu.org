JavaScript Jasmine Demo ======================= 在命令行中运行`Jasmine`跑测试 ``` npm install npm test ```
spec/ hello-spec.js 
var hello = require('../src/hello');

describe("A suite", function () {
  it("contains spec with an expectation", function () {
    expect(hello()).toBe("Hello,world!");
  });
});
spec/support/ jasmine.json 
{
  "spec_dir": "spec",
  "spec_files": [
    "**/*[sS]pec.js"
  ],
  "helpers": [
    "helpers/**/*.js"
  ],
  "stopSpecOnExpectationFailure": false,
  "random": false
}
How to create the `jasmine.json` -------------------------------- ``` jasmine init ``` It will create `spec/support/jasmine.json` with default content.
src/ core.js 
module.exports = function() {
  console.log("this is core function");
}
src/ hello.js 
const core = require('./core');

module.exports = function () {
  core();
  let h = "Hello,";
  let w = "world!";
  console.log("=========== hello =========");
  return h + w;
};
package.json 
{
  "scripts": {
    "test": "jasmine"
  },
  "devDependencies": {
    "jasmine": "^3.2.0"
  }
