JSON Demo ========= It's very easy to handle JSON in python. ``` python3 json_demo.py python3 json_file_demo.py ```
json_file_demo.py 
import json
import os

data_file = './data.json'
if not os.path.exists(data_file):
    open(data_file, 'a').close()

with open(data_file, 'w') as f:
    json.dump({"name": "Freewind", "age": 1000}, f)

with open(data_file, 'r') as f:
    print(json.load(f))
json_demo.py 
import json

print(json.dumps({
    "name": "Freewind",
    "age": 1000
}))

print(json.loads('{"name": "Freewind", "age": 1000}'))
