TypeScript Check Object is Instance of Class Demo ================================================= Check if an object is instance of some class ``` npm install npx ts-node hello.ts ```
hello.ts 
class A {
}

class B {
}

const x: any = new A()

console.log(x instanceof A)
console.log(x instanceof B)
