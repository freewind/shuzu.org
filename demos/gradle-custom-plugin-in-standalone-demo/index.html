gradle-custom-plugin-in-standalone-demo
Gradle Custom Plugin Defined in `build.gradle` Demo =================================================== Publish the plugin to maven local: ``` ./gradlew install ``` Then use in testing project, see `testing-project/build.gradle`. Test: ``` cd testing-project ./gradlew hello ```
build.gradle
build.gradle
apply plugin: 'groovy'
apply plugin: 'maven'

group 'freewind.demos'
version '0.1.0'

dependencies {
    compile gradleApi()
    compile localGroovy()
}
build.gradle
testing-project/build.gradle
buildscript {
    repositories {
        mavenLocal()
    }
    dependencies {
        classpath 'freewind.demos:gradle-custom-plugin-in-standalone-demo:0.1.0'
    }
}

apply plugin: 'gradle-custom-plugin-in-standalone-demo'

hello {
    message = 'Hello, HelloPlugin'
}
