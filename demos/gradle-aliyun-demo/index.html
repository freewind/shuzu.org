gradle-aliyun-demo
Gradle Aliyun Test ================== Use Aliyun's maven repository with gradle. Run tests: ``` ./gradlew test ``` It should be fast in China. Note ---- Use `--debug` to show more log to see if it's really use aliyun.
build.gradle
build.gradle
apply plugin: 'java'

repositories {
    // comment it for demo
    // mavenLocal()
	maven {
        url 'http://maven.aliyun.com/nexus/content/groups/public/'
    }
}

dependencies {
    compile(
            'org.apache.commons:commons-lang3:3.4'
    )
    testCompile(
            'junit:junit:4.11',
            'org.easytesting:fest-assert-core:2.0M10'
    )
}

// used to generate `./gradlew`
// by run `gradle wrapper`
task wrapper(type: Wrapper) {
   gradleVersion = '2.4' //version required
}
Hello.java
src/main/java/my/Hello.java
package my;

public class Hello {
    public String hello(String name) {
        return "Hello, " + name;
    }
}
HelloSpec.java
src/test/java/my/HelloSpec.java
package my;

import org.junit.Test;

import static org.fest.assertions.api.Assertions.assertThat;

public class HelloSpec {

    @Test
    public void shouldSayHello() throws Exception {
        String greeting = new Hello().hello("gradle");
        assertThat(greeting).isEqualTo("Hello, gradle");
    }

}
