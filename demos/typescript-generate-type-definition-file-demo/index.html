TypeScript Hello World Demo ============================ ``` npm install npx tsc --declaration hello.ts ``` 将会生成两个文件： - `hello.js`: 编译成的js文件，不是我们关注的重点 - `hello.d.ts`: 生成的类型说明文件，是我们需要的

class Point {
    static origin = new Point(0, 0);

    x: number;
    
    constructor(x: number, public y: number = 0) {
        this.x = x;
    }

    dist() {
        return Math.sqrt(this.x * this.x + this.y * this.y);
    }

}
