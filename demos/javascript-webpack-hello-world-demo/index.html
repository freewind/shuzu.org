JavaScript Webpack Hello World Demo =================================== ``` npm install npm start ``` Then open `index.html` in browser ![demo](./images/demo.jpg)
content.js
content.js
module.exports = 'Hello Webpack (from content.js)'
entry.js
entry.js
let content = require('./content');

document.write(content);
index.html
index.html
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>Webpack Demo</title>
</head>
<body>
	<script src="./bundle.js"></script>
</body>
</html>
package.json
package.json
{
  "scripts": {
    "start": "webpack ./entry.js -o ./bundle.js && open index.html"
  },
  "devDependencies": {
    "webpack": "^4.17.1",
    "webpack-cli": "^3.1.0"
  }
}
