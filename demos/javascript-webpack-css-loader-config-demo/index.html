Webpack Css Loader Config Demo ============================== ``` npm install npm run demo ``` Then open `index.html`. ![demo](./images/demo.jpg)
entry.js
entry.js
require('./index.css')

document.write('page should have a yellow background')
index.css
index.css
body {
  background-color: yellow;
}
webpack.config.js
webpack.config.js
module.exports = {
    entry: './entry.js',
    output: {
        path: __dirname,
        filename: 'bundle.js'
    },
    module: {
        rules: [{
            test: /\.css$/,
            use: [
                {loader: 'style-loader'},
                {loader: 'css-loader'}
            ]
        }]
    }
}
package.json
package.json
{
  "scripts": {
    "demo": "webpack && open index.html"
  },
  "dependencies": {
    "css-loader": "^1.0.0",
    "style-loader": "^0.23.0"
  },
  "devDependencies": {
    "webpack": "^4.17.1",
    "webpack-cli": "^3.1.0"
  }
}

index.html
