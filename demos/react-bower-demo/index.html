React Hello World Demo ====================== ``` npm install -g bower bower install ``` Then open `index.html` in your browser.
bower.json 
{
  "name": "react-hello-world-demo",
  "authors": [
    "Peng.Li <peng.li@rea-group.com>"
  ],
  "description": "",
  "main": "",
  "license": "MIT",
  "homepage": "",
  "ignore": [
    "**/.*",
    "node_modules",
    "bower_components",
    "test",
    "tests"
  ],
  "dependencies": {
    "babel-standalone": "^6.10.3",
    "react": "^15.3.0"
  }
}
index.html 
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8"/>
    <title>Hello React!</title>
    <script src="./bower_components/react/react.min.js"></script>
    <script src="./bower_components/react/react-dom.min.js"></script>
    <script src="./bower_components/babel-standalone/babel.min.js"></script>
</head>
<body>
<div id="content"></div>
<script type="text/babel">
    let FirstReact = React.createClass({
        render: function () {
            return (<div>
                <h1>Hello world</h1>
            </div>)
        }
    });
    ReactDOM.render(
            <FirstReact />,
            document.getElementById("content")
    );
</script>

</body>
</html>
