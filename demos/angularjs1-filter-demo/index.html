AngularJS1 Filter Demo ====================== Define a filter in [AngularJS1](https://angularjs.org/). ``` npm install open index.html ``` ![demo](./images/demo.jpg)
package.json 
{
  "dependencies": {
    "angular": "^1.7.3"
  }
}
index.html 
<html>
<head>
</head>
<body ng-app="app">
<div>
    <div>
        <label>Name</label>
        <input type="text" ng-model="userName"/>
        <div>{{userName | wave}}</div>
    </div>
</div>
<script src="./node_modules/angular/angular.js"></script>
<script src="./app.js"></script>
</body>
</html>
app.js 
const app = angular.module('app', []);
app.filter('wave', function () {
    return function (data) {
        if (data) {
            return '~~~ ' + data + ' ~~~';
        } else {
            data
        }
    }
});
