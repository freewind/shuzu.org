JavaScript Gulp Demo ==================== Gulp是JavaScript中一个构建工具，特点是通过pipe将各功能串起来，不像Grunt那样是基于配置。 ``` npm install npm run demo ``` 将会在`build`目录下生成相应的文件。
hello.js 
function hello() {
    console.log('Hello, gulp')
}
gulpfile.js 
const gulp = require('gulp')
const minify = require('gulp-minify')

gulp.task('default', function () {
    gulp.src('hello.js')
        .pipe(minify())
        .pipe(gulp.dest('build/'))
})
package.json 
{
  "scripts": {
    "demo": "gulp"
  },
  "devDependencies": {
    "gulp": "^3.9.1",
    "gulp-cli": "^2.0.1",
    "gulp-minify": "^3.1.0"
  }
}
