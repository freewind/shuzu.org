TypeScript Mocha Chai Testing Demo ================================== Use mocha and chai to write tests with typescript. ``` npm install npm test ```
hello.ts
hello.ts
export default function hello(name: string): string {
    return `Hello, ${name}!`
}
hello-test.ts
hello-test.ts
import hello from './hello'
import {expect} from 'chai'

describe('hello', () => {
    it('should give hello words to given name', () => {
        const words = hello('typescript')
        expect(words).to.equal('Hello, typescript!')
    })
})
