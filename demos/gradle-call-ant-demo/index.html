Gradle Call Ant Task Demo ========================= How to call ant task in gradle directly Run: ``` ./gradlew hello ```
build.gradle 
task hello() {
    project.ant.echo(message: "Hello, ant")
}
src/test/java/my/ HelloSpec.java 
package my;

import org.junit.Test;

import static org.fest.assertions.api.Assertions.assertThat;

public class HelloSpec {

    @Test
    public void shouldSayHello() throws Exception {
        String greeting = new Hello().hello("gradle");
        assertThat(greeting).isEqualTo("Hello, gradle");
    }

}
src/main/java/my/ Hello.java 
package my;

public class Hello {
    public String hello(String name) {
        return "Hello, " + name;
    }
}
