JavaScript Eslint Demo ====================== ``` npm install npm run demo ``` 将会检查出代码中不符合规范的地方。
``` $ npx eslint index.js ``` ``` /Users/freewind/workspace/javascript-eslint-demo/index.js 1:10 error 'hello' is defined but never used no-unused-vars 3:5 error Unexpected console statement no-console 4:9 error 'b' is assigned a value but never used no-unused-vars ✖ 3 problems (3 errors, 0 warnings) ```
.eslintrc.js 
module.exports = {
    "env": {
        "node": true
    },
    "extends": "eslint:recommended",
    "parserOptions": {
        "ecmaVersion": 5
    },
    "rules": {
        "indent": [
            "error",
            4
        ],
        "linebreak-style": [
            "error",
            "unix"
        ],
        "quotes": [
            "error",
            "single"
        ],
        "semi": [
            "error",
            "never"
        ]
    }
};
index.js 
function hello() {
    var a = 123
    console.log(a)
    var b = a + 'abc'
}
