ScalaFX Hover Demo ==================== Open it in IDEA, and run, you will see a window, and when you hover on the Green block it will turn to red. This code is copied from <http://www.scalafx.org/>
project/ build.properties 
sbt.version=1.0.4
build.sbt 
name := "ScalaFX Hover Demo"

version := "1.0"

organization := "org.my"

scalaVersion := "2.12.4"

sbtVersion := "1.0.4"

libraryDependencies ++= Seq(
  "org.scalafx" %% "scalafx" % "8.0.144-R12"
)
src/main/scala/my/ Hello.scala 
package my

import scalafx.application
import scalafx.application.JFXApp
import scalafx.scene.Scene
import scalafx.scene.control.Label

object Hello extends JFXApp {

  stage = new application.JFXApp.PrimaryStage {
    title.value = "Hello"
    width = 600
    height = 400
    scene = new Scene {
      content = new Label("Hello, JavaFX")
    }
  }
}
