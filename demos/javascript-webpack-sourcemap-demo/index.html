Webpack SourceMap Demo ====================== Generate source-map for the generated bundle.js which is friendly to development. ``` npm install -g webpack npm install webpack ``` Then open `index.html` in your browser, and open the dev tool of the browser, and refresh. You will see the break point and the clear source code. ![webpack-debug](images/webpack-debug.png)
entry.js 
require('./index.css');

debugger;

document.write('it works!');
index.css 
body {
  background-color: yellow;
}
webpack.config.js 
module.exports = {
  entry: './entry.js',
  output: {
    path: __dirname,
    filename: 'bundle.js'
  },
  devtool: 'inline-source-map',
  module: {
    loaders: [{ 
      test: /\.css$/,
      loader: 'style!css' // note style is before css
    }]
  }
}
package.json 
{
  "name": "webpack-config-demo",
  "version": "1.0.0",
  "description": "",
  "main": "index.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "keywords": [],
  "author": "",
  "license": "ISC",
  "dependencies": {
    "css-loader": "^0.23.1",
    "style-loader": "^0.13.1"
  }
}
