Gradle Hello World Demo ======================= A very simple java project with gradle. Run tests: ``` ./gradlew test ``` Update Gradle Version --------------------- ``` gradle wrapper --gradle-version 4.8.1 ./gradlew -version ``` Setup in IDEA -------------- Use command line: ``` idea . ``` Or import the project directly with IDEA.
build.gradle
build.gradle
apply plugin: 'java'

repositories {
    mavenLocal()
    maven { url 'http://maven.aliyun.com/nexus/content/groups/public/' }
    maven { url 'http://maven.aliyun.com/nexus/content/repositories/jcenter' }
    jcenter()
}

dependencies {
    compile 'org.apache.commons:commons-lang3:3.4'
    testCompile 'junit:junit:4.11'
    testCompile 'org.easytesting:fest-assert-core:2.0M10'
}

apply from: 'tasks.gradle'
Hello.java
src/main/java/my/Hello.java
package my;

public class Hello {
    public String hello(String name) {
        return "Hello, " + name;
    }
}
HelloSpec.java
src/test/java/my/HelloSpec.java
package my;

import org.junit.Test;

import static org.fest.assertions.api.Assertions.assertThat;

public class HelloSpec {

    @Test
    public void shouldSayHello() throws Exception {
        String greeting = new Hello().hello("gradle");
        assertThat(greeting).isEqualTo("Hello, gradle");
    }

}
tasks.gradle
tasks.gradle
task hello() {
    println("Hello from another file")
}
