Jest Demo ========== Use jest to mock modules, and put test files in `spec`(not default `__tests__`) ``` npm install -g jest-cli npm install npm test ``` Notice `jest` configuration in `package.json`
square-sum-spec.js
spec/square-sum-spec.js
jest.unmock('../square-sum');

describe('square-sum', () => {
  it('[1,2,5]', () => {
    jest.mock('../square', () => {
      return jest.fn((n) => n * n);
    });
    const squareSum = require('../square-sum');
    const result = squareSum([1, 2, 5]);
    expect(result).toEqual(30);
  })
})

square-sum.js
const square = require('./square');

function squareSum(numbers) {
  return numbers.reduce((r, n) => r + square(n));
}

module.exports = squareSum;
