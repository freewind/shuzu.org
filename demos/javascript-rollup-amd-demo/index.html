JavaScript Rollup to AMD Demo ============================= Rollup使用AMD模块系统编译JS代码 ``` npm install npm run demo ``` 将生成`bundle.js`文件
hello.js 
import {capitalize} from 'lodash'

export default function (name) {
    return 'Hello, ' + capitalize(name) + '!'
}
编译出来的`bundle.js` ---------------- ``` define(['lodash'], function (lodash) { 'use strict'; function hello (name) { return 'Hello, ' + lodash.capitalize(name) + '!' } hello('rollup'); }); ```
package.json 
{
  "scripts": {
    "demo": "rollup -c"
  },
  "dependencies": {
    "lodash": "^4.17.10"
  },
  "devDependencies": {
    "rollup": "^0.65.0"
  }
}
rollup.config.js 
export default {
    input: 'main.js',
    output: {
        file: 'bundle.js',
        format: 'amd',
    },
    external: ['lodash']
}
main.js 
import hello from './hello'

hello('rollup')
