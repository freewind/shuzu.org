Gradle Custom Plugin Defined in `build.gradle` Demo =================================================== ``` ./gradlew hello ``` Resources: --------- - <https://docs.gradle.org/current/userguide/custom_plugins.html>
build.gradle
build.gradle
class HelloPluginExtension {
    String message = 'Hello from HelloPlugin'
}

class HelloPlugin implements Plugin<Project> {
    void apply(Project project) {
        HelloPluginExtension extension = project.extensions.create('hello', HelloPluginExtension)
        project.task('hello') {
            doLast {
                println(extension.message)
            }
        }
    }
}

apply plugin: HelloPlugin

// hello.message = "configured hello"
// or
hello {
    message = "configured hello"
}
