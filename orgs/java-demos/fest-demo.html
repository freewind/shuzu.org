Gradle Test ========== A very simple java project with gradle. Run --- ./gradlew run Will print `HELLO, WORLD!`
build.gradle
build.gradle
apply plugin: 'java'
apply plugin: 'idea'

apply plugin:'application'
mainClassName = "my.Hello"

repositories {
    mavenCentral()
}

dependencies {
    compile(
            'org.apache.commons:commons-lang3:3.2.1',
            'org.apache.commons:commons-io:1.3.2',
            'com.google.guava:guava:16.0'
    )
    testCompile(
        'junit:junit:4.11'
    )
}

task wrapper(type: Wrapper) {
    gradleVersion = '1.11';
}
Hello.java
src/main/java/my/Hello.java
package my;

import org.apache.commons.lang3.StringUtils;

public class Hello {
    public static void main(String[] args) {
        System.out.println(StringUtils.upperCase("Hello, world!"));
    }
}
