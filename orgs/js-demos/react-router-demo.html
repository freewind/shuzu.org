Webpack Router Demo ======================== Put all settings in `webpack.config.js` ``` npm install -g webpack npm install webpack ``` Then open `index.html` in your browser.
index.css
index.css
body {
  background-color: yellow;
}
index.html
index.html
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>Webpack Router Demo</title>
</head>
<body>
	<div id="content"></div>
	<script src="bundle.js"></script>
</body>
</html>
webpack.config.js
webpack.config.js
module.exports = {
  entry: './app/entry.jsx',
  output: {
    path: __dirname,
    filename: 'bundle.js'
  },
  module: {
    loaders: [{ 
      test: /\.(js|jsx)$/,
      exclude: /node_modules/,
      loader: 'babel',
      query: {
        presets: ['es2015', 'react']
      }
    }]
  }
}
