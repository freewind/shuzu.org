Jasmine Debugging Demo ====================== How to use webstorm to debug jasmine specs, especially with the es6 javascript code. ``` npm install ``` Then use your webstorm to open this project, set some break points, then you can single debug in webstorm by running `run-specs.js` directly in debugging mode. Notice ------ 1. You must upgrade webstorm to latest version(2016.02) to make it run successfully. [Download link](https://www.jetbrains.com/webstorm/download/). 2. We can't use `import` and `export` keywords in JavaScript, since they are not working easily without `babel`, instead, we should use `require()` and `module.exports`.
run-spec.js
run-spec.js
var jasmine = require('jasmine-node');
jasmine.executeSpecsInFolder({
  specFolders: ['spec'],
  isVerbose: true,
  showColors: true
});
hello-spec.js
spec/hello-spec.js
var Hello = require('../src/hello');

describe("A suite", function () {
  it("contains spec with an expectation", function () {
    const hello = new Hello();
    expect(hello.greeting()).toBe("Hello, Freewind!!");
  });
});
hello.js
src/hello.js
const User = require('./user');

module.exports = class Hello {
  greeting() {
    const user = new User('Freewind');
    return 'Hello, ' + user.nameBang();
  }
};
user.js
src/user.js
module.exports =  class User {
  constructor(name) {
    this.name = name;
  }

  nameBang() {
    const bang = this.name + '!!';
    return bang;
  }
}
