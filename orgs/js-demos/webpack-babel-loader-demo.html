Webpack ES6 import/export Demo =============================== ``` npm install -g webpack npm install webpack ``` Then open `index.html` in your browser. You will see the `import/export` syntax be converted to normal javascript code by `babel`.
core.js
core.js
class Core {
  name() {
    return 'imported from core module';
  }
}

export default Core;
entry.js
entry.js
import Core from './core';

const name = new Core().name();
document.write(name);
index.html
index.html
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>Webpack babel-loader Demo</title>
</head>
<body>
	<script src="bundle.js"></script>
</body>
</html>
webpack.config.js
webpack.config.js
module.exports = {
  entry: './entry.js',
  output: {
    path: __dirname,
    filename: 'bundle.js'
  },
  module: {
    loaders: [{ 
      test: /\.js$/,
      exclude: /node_modules/,
      loader: 'babel',
      query: {
        presets: ['es2015']
      }
    }]
  }
}
